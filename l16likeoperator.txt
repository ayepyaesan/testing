Wildcards ( 7 types)

WHERE customername LIKE 'a%' Finds any value that starts with 'a'
WHERE customername LIKE '%a' Finds any value that ends with 'a'
WHERE customername LIKE '%or%' Finds any value that has 'or' in any position
WHERE customername LIKE '_b' Finds any value that has 'b' in the second position
WHERE customername LIKE "a_%" Finds any value that starts with 'a' and at least 2 characters in length
WHERE customername LIKE "a__%" Finds any value that starts with 'a' and at least 3 characters in length
WHERE customername LIKE "a%o" Finds any value that starts with 'a' and ends with 'o'

Syntax

Method 1
SELECT * FROM table_name
WHERE column_name LIKE pattern;

Method 2
SELECT column1,column2,...
FROM table_name
where column_name LIKE pattern;
